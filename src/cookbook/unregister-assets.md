# How to Unregister an Asset

```rust
fn unregister_asset(
    iroha: &Client,
) {
    let roses_of_alice = "rose##alice@wonderland".parse::<AssetId>().unwrap();
    let unregister_roses_of_alice = Unregister::asset(roses_of_alice);
    iroha.submit(unregister_roses_of_alice).unwrap();
}
```
